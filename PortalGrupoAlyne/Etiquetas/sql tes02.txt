select CAB.CODPARCTRANSP
	, TRA.RAZAOSOCIAL AS TRANSPORTADORA
	, CLI.RAZAOSOCIAL AS CLIENTE
	, CID.NOMECID AS MUNICIPIO
	, CAB.AD_UF AS UF
	, CAB.NUMNOTA AS NF
	, CAB.QTDVOL AS VOLUME
	, CAB.DTNEG AS DATA_SAIDA
	, CAB.OBSERVACAO AS PEDIDO

FROM TGFCAB CAB
LEFT JOIN TGFPAR TRA ON TRA.CODPARC = CAB.CODPARCTRANSP
LEFT JOIN TGFPAR CLI ON CLI.CODPARC = CAB.CODPARC
LEFT JOIN TSICID CID ON CID.CODCID  = CLI.CODCID
WHERE CAB.NUMNOTA = $NUNOTA$
AND CAB.TIPMOV = 'V'
